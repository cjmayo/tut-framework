29 std::string escape(const std::string& orig)
30 {

32  std::string::const_iterator i,e;
33  i = orig.begin();
34  e = orig.end();
35 
36  while (i != e)
37  {
38  if ((*i >= 'a' && *i <= 'z') ||
39  (*i >= 'A' && *i <= 'Z') ||
40  (*i >= '0' && *i <= '9') )
41  {

64  e = orig.end();

84  }

87  }

90  }

136 {
137  std::getline(is,tr.group);
138  if (is.eof())

143 
144  tr.test = -1;
